342  zw = grid%VARS%zw
343  psic = grid%SOIL%psic
344  isoil = grid%SOIL%isoil
345  zrzmax = global%zrzmax
346  tzsm1 = grid%VARS%tzsm1
347  thetas = grid%SOIL%thetas
348  rzsm1 = grid%VARS%rzsm1
349  zbar1 = cat%zbar1
350  pixsiz = global%pixsiz
351 
352 ! ====================================================================
353 ! Chose option for calculating baseflow.
354 ! ====================================================================
355 
356  if (iopbf.eq.0) then
357 
358 ! --------------------------------------------------------------------&
359 ! Calculate baseflow according to Sivapalan et al.(1987).
360 ! --------------------------------------------------------------------&
361 
362  qb = q0 * dexp(-ff*zbar)
363 
364  else
365 ! --------------------------------------------------------------------&
366 ! Calculate baseflow according to Troch et al.(1992).
367 ! --------------------------------------------------------------------&
368 
369  hbar = dtil - zbar
370  qb = 5.772*basink*hbar*hbar*dd*xlength
371 
372  endif
373 
374 ! ====================================================================
375 ! Determine net recharge to water table and available soil
376 ! water storage.
377 ! ====================================================================
378 
379  zbrflx = (gwtsum - capsum - etwtsum - (qb/ area)) * dt
380  zbrpor = (rzpsum+tzpsum) * (zbar-psicav)
381 
382  if (zbrflx.gt.zbrpor) then
383 
384 ! ====================================================================
385 ! If net recharge exceeds storage assign overflow to streamflow.
386 ! ====================================================================
387 
388  qzbar = (zbrflx - zbrpor)/dt
389  zbrflx = zbrpor
390  qb = qb + qzbar*area
391 
392 ! --------------------------------------------------------------------&
393 ! If water table is falling but storage is full zbar1 will
394 ! blow up. use rzsm1 and tzsm1 (vs rzsm and tzsm) to
395 ! calculate storage so that it is > zero.
396 ! --------------------------------------------------------------------&
397 
398  else if ((zbrflx.le.zero).and.(zbrpor.le.zero)) then
399 
400 ! --------------------------------------------------------------------&
401 ! Recalculate rzpsum and tzpsum with new soil moistures.
402 ! --------------------------------------------------------------------&
403 
404  rzpsum = zero
405  tzpsum = zero
406 
407  do 100 mm=1,npix
408 
409  if (ivgtyp(ilandc(mm)).ge.0) then
410 
411  if ((icatch(mm)).eq.ic) then
412 
413  if ((zw(mm)-psic(isoil(mm))).gt.zrzmax) then
414 
415  tzpsum = tzpsum+(thetas(isoil(mm))-tzsm1(mm))
416 
417  else if ((zw(mm)-psic(isoil(mm)).gt.zero)) then
418 
419  rzpsum = rzpsum+(thetas(isoil(mm))-rzsm1(mm))
420 
421  endif
422 
423  endif
424 
425  endif
426 
427 100 continue
428 
429  endif
430 
431 ! ====================================================================
432 ! Update zbar by taking the total flux and dividing by
433 ! the average porosity just above the water table.
434 ! ====================================================================
435 
436 ! --------------------------------------------------------------------&
437 ! If the available porosity is nonzero divide the flux by its value.
438 ! --------------------------------------------------------------------&
439 
440  if ( (rzpsum+tzpsum).gt.(0.001d0)) then
441 
442  zbar1 = zbar - zbrflx/(rzpsum+tzpsum)
443 
444  endif
445 
446  if ( (rzpsum+tzpsum).le.(0.001d0)) then
447 
448  zbar1=zbar
449 
450  endif
451 
452 ! ====================================================================
453 ! Find change in water table.
454 ! ====================================================================
455 
456  dzbar = zbar1-zbar
457 
458 ! ====================================================================
459 ! Find new average water table depth and baseflow for entire region.
460 ! ====================================================================
461 
462  qbreg = qbreg + qb
463  zbar1rg = zbar1rg + zbar1*area/(pixsiz*pixsiz)
464 
465 ! ====================================================================
466 ! Format statements.
467 ! ====================================================================
468 
469  cat%zbar1 = zbar1
470  cat%qb = qb
471 
472  return
473 
474  end subroutine upzbar
