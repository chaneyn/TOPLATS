873 ! subroutine petpen
874 !
875 ! ====================================================================
876 !
877 ! Subroutine to calculate the potential evapotranspiration
878 ! using penman's equation for bare soil and penman-monteith
879 ! for vegetated areas.
880 !
881 ! ====================================================================
882 !
883 ! NOTE: sign convention: all radiative fluxes directed toward the
884 ! surface are positive (e.g. net radiation). All
885 ! non-radiative fluxes directed away from surface are
886 ! positive (e.g. latent, sensible and soil heat fluxes)
887 !
888 ! ====================================================================
889 
890  subroutine petpen(GRID_VEG,GRID_MET,GRID_VARS,tcel,vpsat,vpdef,f1par,&
891  albd,xlai,rsd,rsmin,rsmax,rpl,tkel,vppa,f3vpd,f3vpdpar,f4temp,trefk,&
892  f4temppar,rnetpn,gbspen,rnetd,rnetw,gd,gw,rescan,ravd,xlhv,&
893  row,epetd,epetw,ravw,psychr,xled,xlew,hd,hw,cp,roa)
894 
895  implicit none
896  include "help/petpen.h"
897 
898  type (grid_met_template) :: grid_met
899  type (grid_veg_template) :: grid_veg
900  type (grid_vars_template) :: grid_vars
901 
902  albd = grid_veg%albd
903  xlai = grid_veg%xlai
904  rsd = grid_met%rsd
905  rsmin = grid_veg%rsmin
906  rsmax = grid_veg%rsmax
907  rpl = grid_veg%Rpl
908  trefk = grid_veg%trefk
909  rnetpn = grid_vars%rnetpn
910  gbspen = grid_vars%gbspen
911  rescan = grid_veg%rescan
912  row = grid_vars%row
913  cp = grid_vars%cp
914 
915 
916 
917 ! ====================================================================
918 ! Calculate the atmospheri! vapor pressure deficit and slope of
919 ! vapor pressure-temperature curve.
920 ! ====================================================================
921 
922  vpsat = 611.d0 * dexp((17.27d0*tcel)/(237.3d0+tcel))
923  vpdef = vpsat - vppa
924  dvpsdt = 4098.d0 * vpsat/((237.3d0+tcel)**two)
925 
926 ! ====================================================================
927 ! Calculate the environmental controls on stomatal
928 ! ====================================================================
929 
930 ! --------------------------------------------------------------------
931 ! Photosynthetically active radiation (PAR).
932 ! --------------------------------------------------------------------
933 
934 
935  f1par = 1
936 
937 ! --------------------------------------------------------------------
938 ! Vapor pressure deficit.
939 ! --------------------------------------------------------------------
940  f3vpd = clcf3vpd(tkel,vppa,f3vpdpar)
941 
942 ! --------------------------------------------------------------------
943 ! Air temperature.
944 ! --------------------------------------------------------------------
945 
946  f4temp = clcf4temp(tkel,trefk,f4temppar)
947 
948 ! ====================================================================
949 ! Assign inputs to each net radiation variable and ground heat flux.
950 ! ====================================================================
951 
952  rnetd = rnetpn
953  rnetw = rnetpn
954  gd = gbspen
955  gw = gbspen
956 
957 ! ====================================================================
958 ! Calculate potential evapotranspiration rate from dry canopy or
959 ! bare soil.
960 ! ====================================================================
961 
962  pstar = psychr * (one+f1par*f3vpd*f4temp*rescan/ravd)
963 
964  if (1.eq.0) then
965 
966 ! --------------------------------------------------------------------
967 ! This is the original formulation without environmental control
968 ! on stomatal resistance that we do not want to reach.
969 ! --------------------------------------------------------------------
970  epetdmf = ((dvpsdt*(rnetd-gd)) +&
971  ((cp*roa*vpdef)/ravd)) /&
972  ((dvpsdt+pstar)*xlhv)
973 
974  endif
975 
976  epetdmf = ((dvpsdt*(rnetd-gd)) +&
977  ((cp*roa*vpdef)/&
978  (f1par*f3vpd*f4temp*rescan+ravd))) /&
979  ((dvpsdt+pstar)*xlhv)
980 
981 
982  epetd=epetdmf/row
983 
984 ! ====================================================================
985 ! Calculate potential evaporation rate from wet canopy.
986 ! ====================================================================
987 
988  epetwmf = ((dvpsdt*(rnetw-gw)) +&
989  ((cp*roa*vpdef)/ravw)) /&
990  ((dvpsdt+psychr)*xlhv)
991  epetw = epetwmf/row
992 
993 ! ====================================================================
994 ! Calculate potential latent heat fluxes.
995 ! ====================================================================
996 
997  xled = epetdmf*xlhv
998  xlew = epetwmf*xlhv
999 
1000 ! ====================================================================
1001 ! Calculate sensible heat fluxes using potential evapotranspiration
1002 ! rate.
1003 ! ====================================================================
1004 
1005  hd = rnetd-xled-gd
1006  hw = rnetw-xlew-gw
1007 
1008  grid_veg%albd = albd
1009  grid_veg%xlai = xlai
1010  grid_met%rsd = rsd
1011  grid_veg%rsmin = rsmin
1012  grid_veg%rsmax = rsmax
1013  grid_veg%Rpl = rpl
1014  grid_veg%trefk = trefk
1015  grid_vars%rnetpn = rnetpn
1016  grid_vars%gbspen = gbspen
1017  grid_veg%rescan = rescan
1018  grid_vars%row = row
1019  grid_vars%cp = cp
1020 
1021  return
1022 
1023  end subroutine petpen

1613 !
1614 ! function clcf1par
1615 !
1616 ! ====================================================================
1617 !
1618 ! Calculate the limiting factor due to Photosynthetically
1619 ! Active Radiation (PAR) for canopy resistance following
1620 ! Jarvis (1976), Dickenson et al. (1986), Noilhan and Planton (1989)
1621 !
1622 !
1623 ! f1par = + f
1624 ! --------------------
1625 ! + rsmin/rsmax
1626 !
1627 ! where
1628 ! = 0.55 * (1-alb)*Rg * 2
1629 ! --- ---
1630 ! Rgl LAI
1631 !
1632 !
1633 ! The values for rsmax, Rgl are usually from Dickenson et al. (1986).
1634 ! Units:
1635 ! rsmax [s/m] = 5000
1636 ! Rgl [w/m^2] = 30 for forest and 100 for crop/grassland
1637 !
1638 ! ====================================================================
1639 
1640  function clcf1par(alb,LAI,Rg,rsmin,rsmax,Rgl)
1641  implicit none
1642  include "help/clcf1par.h"
1643  !type (GRID_VEG_template) :: GRID_VEG
1644  data a1/0.19/,a2/1128/,a3/30.8/
1645 
1646  par = 0.55 * (1.d0 - alb)*rg
1647 
1648  f = 2.d0*par/(rgl*lai)
1649 
1650  f1par = (1.d0 + f)/(f + rsmin/rsmax)
1651 
1652  clcf1par = f1par
1653 
1654  return
1655 
1656  end function clcf1par

1662 ! ====================================================================
1663 ! Calculate the limiting factor due to vapor pressure
1664 ! for canopy resistance following
1665 ! Jarvis (1976), Dickenson et al. (1986), Noilhan and Planton (1989)
1666 !
1667 !
1668 ! f3vpd = 1
1669 ! ----------------------
1670 ! 1.d0 - g*(esat(Ts)-ea)
1671 !
1672 !
1673 !
1674 ! Revision 06/25/97: added limits to vpd of 1/g (max) and 0 (min)
1675 ! by Ted Endreny and Mark Zion.
1676 !
1677 ! ====================================================================
1678 
1679  function clcf3vpd(Ts,ea,g)
1680  implicit none
1681  include "help/clcf3vpd.h"
1682 
1683  rmax_vpd = (1.d0/g) - 0.1
1684  rmin_vpd = 0.0
1685  vpd = esat(ts)-ea
1686  if (vpd.lt.rmin_vpd) vpd=rmin_vpd
1687  if (vpd.gt.rmax_vpd) vpd=rmax_vpd
1688 
1689  if ( (1.d0 - g*(esat(ts)-ea)).ge.0.25d0) then
1690 
1691  f3vpd = 1.d0/(1.d0 - g*vpd)
1692 
1693  else
1694 
1695  f3vpd = 4.d0
1696 
1697  endif
1698 
1699  clcf3vpd = f3vpd
1700 
1701  return
1702  end function clcf3vpd

1706 ! function clcf4temp
1707 !
1708 ! ====================================================================
1709 !
1710 ! Calculate the limiting factor due to air temperature
1711 ! for canopy resistance following
1712 ! Jarvis (1976), Dickenson et al. (1986), Noilhan and Planton (1989)
1713 !
1714 !
1715 ! f4temp =___________1______________
1716 ! 1.0 - f4par(tref - tair)^2
1717 !
1718 ! where
1719 ! f4par is usually 0.0016
1720 ! tref and tpar in Kelvin or Celsius
1721 !
1722 ! Revision 6/25/97: added limits on air temperature to:
1723 ! maximum: tref + (1/f4par)^.5 - 0.01
1724 ! minimum: tref - (1/f4par)^.5 + 0.01
1725 ! by Ted Endreny and Mark Zion
1726 !
1727 ! Revision 7/22/98: add further limits to air temperature in
1728 ! order to reduce the maximum f4temp. 0.01
1729 ! is increased to 5.
1730 !
1731 ! ====================================================================
1732 
1733  function clcf4temp(tair,tref,f4par)
1734 
1735  implicit none
1736  include "help/clcf4temp.h"
1737 
1738 ! ====================================================================
1739 ! Set maximum and minimum values for air temperature so
1740 ! factor does not go negative.
1741 ! ====================================================================
1742 
1743  rmin_tair = tref - sqrt(1/f4par) + 0.01
1744  rmax_tair = tref + sqrt(1/f4par) - 0.01
1745 
1746  rmin_tair = tref - sqrt(1/f4par) + 5.
1747  rmax_tair = tref + sqrt(1/f4par) - 5.
1748 
1749  if (tair.lt.rmin_tair) then
1750 
1751  tair_calc = rmin_tair
1752 
1753  else if (tair.gt.rmax_tair) then
1754 
1755  tair_calc = rmax_tair
1756 
1757  else
1758 
1759  tair_calc = tair
1760 
1761  endif
1762 
1763 ! ====================================================================
1764 ! Calculate temperature factor.
1765 ! ====================================================================
1766 
1767  f4temp = 1.d0/(1.d0 - f4par*(tref - tair_calc)**2)
1768 
1769  clcf4temp = f4temp
1770 
1771  return
1772 
1773  end function clcf4temp
