1386 ! function clcf1par
1387 !
1388 ! ====================================================================
1389 !
1390 ! Calculate the limiting factor due to Photosynthetically
1391 ! Active Radiation (PAR) for canopy resistance following
1392 ! Jarvis (1976), Dickenson et al. (1986), Noilhan and Planton (1989)
1393 !
1394 !
1395 ! f1par = + f
1396 ! --------------------
1397 ! + rsmin/rsmax
1398 !
1399 ! where
1400 ! = 0.55 * (1-alb)*Rg * 2
1401 ! --- ---
1402 ! Rgl LAI
1403 !
1404 !
1405 ! The values for rsmax, Rgl are usually from Dickenson et al. (1986).
1406 ! Units:
1407 ! rsmax [s/m] = 5000
1408 ! Rgl [w/m^2] = 30 for forest and 100 for crop/grassland
1409 !
1410 ! ====================================================================
1411 
1412  function clcf1par(alb,LAI,Rg,rsmin,rsmax,Rgl)

1418 
1419  f = 2.d0*par/(rgl*lai)
1420 
1421  f1par = (1.d0 + f)/(f + rsmin/rsmax)
1422 
1423  clcf1par = f1par
1424 
1425  return
1426 
1427  end function clcf1par

1453

1455  rmin_vpd = 0.0

1458  if (vpd.gt.rmax_vpd) vpd=rmax_vpd
1459 
1460  if ( (1.d0 - g*(esat(ts)-ea)).ge.0.25d0) then
1461 
1462  f3vpd = 1.d0/(1.d0 - g*vpd)
1463 
1464  else
1465 
1466  f3vpd = 4.d0
1467 
1468  endif
1469 
1470  clcf3vpd = f3vpd
1471 
1472  return
1473  end function clcf3vpd

1480 !

1487 ! 1.0 - f4par(tref - tair)^2

1494 ! maximum: tref + (1/f4par)^.5 - 0.01
1495 ! minimum: tref - (1/f4par)^.5 + 0.01
1496 ! by Ted Endreny and Mark Zion
1497 !
1498 ! Revision 7/22/98: add further limits to air temperature in
1499 ! order to reduce the maximum f4temp. 0.01
1500 ! is increased to 5.
1501 !
1502 ! ====================================================================

1508  real*8 f4temp,clcf4temp

1516  rmax_tair = tref + sqrt(1/f4par) - 0.01

1518  rmin_tair = tref - sqrt(1/f4par) + 5.
1519  rmax_tair = tref + sqrt(1/f4par) - 5.

1522 
1523  tair_calc = rmin_tair

1532 
1533  endif

1535 ! ====================================================================

1540
